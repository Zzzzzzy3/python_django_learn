# 字典示例
my_dict = {"name": "Alice", "age": 25, "city": "New York"}
print(my_dict["name"])  # 输出：Alice
my_dict["email"] = "alice@example.com"  # 添加新键值对
del my_dict["age"]  # 删除键值对

# 集合示例
my_set = {1, 2, 3, 4}
my_set.add(5)  # 添加元素
my_set.remove(2)  # 删除元素
print(my_set & {3, 4, 5})  # 输出：{3, 4, 5}的交集 {3,4,5}与原集合{1,3,4,5}的交集是{3,4,5}原集合是{1,2,3,4}
